### API Authentication

<CodeGroup>

```typescript TypeScript
// Login and get session token
const loginResponse = await fetch('/api/auth/login', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
  },
  body: JSON.stringify({
    email: 'user@example.com',
    password: 'securePassword123'
  })
});

const { token } = await loginResponse.json();

// Use token for authenticated requests
const apiResponse = await fetch('/api/ai/generate', {
  method: 'POST',
  headers: {
    'Authorization': `Bearer ${token}`,
    'Content-Type': 'application/json',
  },
  body: JSON.stringify({
    text: "Original post content",
    tone: "professional"
  })
});
```

```javascript JavaScript
// Login
fetch('/api/auth/login', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
  },
  body: JSON.stringify({
    email: 'user@example.com',
    password: 'securePassword123'
  })
})
.then(response => response.json())
.then(data => {
  const token = data.token;
  
  // Store token for future requests
  localStorage.setItem('authToken', token);
});
```

```python Python
import requests

# Login
login_response = requests.post('https://app.replyiq.com/api/auth/login', 
  json={

# Use token for API calls
headers = {'Authorization': f'Bearer {token}'}
api_response = requests.post('https://app.replyiq.com/api/ai/generate',
  headers=headers,
  json={'text': 'Original post', 'tone': 'professional'}
)
```

## Session Management

### Session Configuration

```typescript
// Session configuration
{
  strategy: "jwt",
  maxAge: 30 * 24 * 60 * 60, // 30 days
  updateAge: 24 * 60 * 60,    // 24 hours
  generateSessionToken: () => crypto.randomUUID(),
}
```

### Session Validation

<CodeGroup>

```typescript TypeScript
// Validate session on server
import { getServerSession } from "next-auth/next";
import { authOptions } from "@/lib/auth";

export async function validateSession(req: Request) {
  const session = await getServerSession(authOptions);
  
  if (!session) {
    throw new Error("Unauthorized");
  }
  
  return session.user;
}
```

```typescript Client-side
// Check session on client
import { useSession } from "next-auth/react";

function ProtectedComponent() {
  const { data: session, status } = useSession();
  
  if (status === "loading") return <p>Loading...</p>;
  if (status === "unauthenticated") return <p>Access Denied</p>;
  
  return <p>Welcome {session.user.email}!</p>;
}
```

</CodeGroup>

## Error Handling

### Common Authentication Errors

<CodeGroup>

```json 401 Unauthorized
{
  "error": "Unauthorized",
  "message": "Invalid or missing authentication token",
  "code": "AUTH_REQUIRED"
}
```

```json 403 Forbidden
{
  "error": "Forbidden", 
  "message": "Insufficient permissions for this operation",
  "code": "INSUFFICIENT_PERMISSIONS"
}
```

```json 429 Rate Limited
{
  "error": "Too Many Requests",
  "message": "Rate limit exceeded. Try again in 60 seconds",
  "code": "RATE_LIMIT_EXCEEDED",
  "retryAfter": 60
}
```

</CodeGroup>

### Error Handling Best Practices

<Tip>
  Always handle authentication errors gracefully and provide clear feedback to users about what went wrong and how to fix it.
</Tip>

<CodeGroup>

```typescript TypeScript
try {
  const response = await fetch('/api/ai/generate', {
    headers: {
      'Authorization': `Bearer ${token}`,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify(requestData)
  });

  if (response.status === 401) {
    // Token expired, redirect to login
    window.location.href = '/auth/login';
    return;
  }

  if (response.status === 429) {
    // Rate limited, show retry message
    const data = await response.json();
    showError(`Rate limited. Try again in ${data.retryAfter} seconds`);
    return;
  }

  const result = await response.json();
  return result;
} catch (error) {
  console.error('Authentication error:', error);
  showError('Authentication failed. Please try again.');
}
```

</CodeGroup>
